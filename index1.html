Setup Google Sheets — 5 Menit

Ikuti langkah berikut. Setelah selesai, setiap laporan yang disimpan akan otomatis masuk ke spreadsheet kamu.

1
Buat Spreadsheet baru
Buka sheets.google.com → klik + untuk buat spreadsheet kosong. Beri nama misalnya "TGOFresh Laporan Panen".
2
Buka Apps Script
Di menu spreadsheet: klik Extensions → Apps Script. Tab baru akan terbuka.
3
Paste kode
Hapus semua kode yang ada di editor, lalu paste kode dari tab "Kode Apps Script" di atas. Klik ikon 💾 Save (Ctrl+S).
4
Deploy sebagai Web App
Klik tombol Deploy → New deployment → pilih type: Web app.
Isi:
 • Execute as: Me
 • Who has access: Anyone
Klik Deploy → izinkan akses jika diminta → salin Web app URL.
5
Sambungkan ke aplikasi
Buka tab "Sambungkan" di atas, paste URL tadi, klik Simpan. Selesai! ✅
⚠️ Penting: Setiap kali kamu mengubah kode Apps Script, kamu perlu deploy ulang (New deployment, bukan Edit deployment) agar perubahan aktif.

💻 Kode Google Apps Script

Copy semua kode di bawah, paste ke editor Apps Script (hapus kode lama dulu).

function doPost(e) {
  try {
    const ss  = SpreadsheetApp.getActiveSpreadsheet();
    const sh  = ss.getSheetByName("Laporan") || ss.insertSheet("Laporan");

    // Buat header jika sheet masih kosong
    if (sh.getLastRow() === 0) {
      sh.appendRow([
        "Tanggal", "Hari", "Jam", "Komoditas",
        "Total Panen (kg)", "Harga/kg (Rp)",
        "Nama Customer", "Qty Customer (kg)",
        "Harga Customer/kg", "Subtotal Customer (Rp)",
        "Total Omzet (Rp)", "Catatan", "Dicatat Pada"
      ]);
      // Format header
      const hdr = sh.getRange(1, 1, 1, 13);
      hdr.setBackground("#1a4d2e");
      hdr.setFontColor("#ffffff");
      hdr.setFontWeight("bold");
      sh.setFrozenRows(1);
    }

    const d = JSON.parse(e.postData.contents);
    const tgl = new Date(d.tanggal + "T00:00:00");
    const days = ["Minggu","Senin","Selasa","Rabu","Kamis","Jumat","Sabtu"];
    const hari = days[tgl.getDay()];
    const timestamp = new Date().toLocaleString("id-ID", {timeZone:"Asia/Jakarta"});

    // Tulis baris per customer
    if (d.customers && d.customers.length > 0) {
      d.customers.forEach(function(c, i) {
        sh.appendRow([
          i === 0 ? d.tanggal : "",
          i === 0 ? hari : "",
          i === 0 ? d.jam : "",
          i === 0 ? d.komoditas : "",
          i === 0 ? d.totalPanen : "",
          i === 0 ? d.hargaKg : "",
          c.nama || "",
          c.qty || "",
          c.harga || "",
          c.subtotal || "",
          i === 0 ? d.omzet : "",
          i === 0 ? (d.catatan || "") : "",
          i === 0 ? timestamp : ""
        ]);
      });
    } else {
      // Tidak ada customer
      sh.appendRow([
        d.tanggal, hari, d.jam, d.komoditas,
        d.totalPanen, d.hargaKg,
        "-", "-", "-", "-",
        d.omzet, d.catatan || "", timestamp
      ]);
    }

    // Auto-resize kolom
    sh.autoResizeColumns(1, 13);

    return ContentService
      .createTextOutput(JSON.stringify({ok: true}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok: false, error: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
